# El estilo profesional de python (`@property`)
## micro ejemplos
```python
class Temperatura:
    def __int__ (self):
        self.__grados = 20

    @property
    def grados(self):
        return self.__grados
    
    @grados.setter
    def grados(self, nuevo_grado):
        if nuevo__grado < 0:
            print('Temperatura invalida')
        else:
            self.__grados = nuevos_grado


clima = Temperatura()

# con el metodo tradicinal necesitamos usar el clima.get_grados() para obtener el valor de grados
# clima._grados NO FUNCIONA

print(clima.grados)

clima.grados = -20

print(clima.grados)
```

# Atributos y metodos de clase
## atributo de clase

```python
class Tienda:
    # atributo de la clase Aplica a todos
    impuesto_iva = 0.13

    def __init__(self, productos):
        self.productos = productos
        
    @classmethod
    def cambiar_impuesto(cls, nuevo_impuesto):
        cls.impuesto_iva = nuevo_impuesto
        
        
Tienda.cambiar_impuesto(0.15)
```
